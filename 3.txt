#include <GL/glut.h>

int spinWin;

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    glBegin(GL_POLYGON);
    glColor3f(1, 0, 0);
    glVertex2f(-200, -100);   // bottom-left

    glColor3f(0, 1, 0);
    glVertex2f(-200, 100);    // top-left

    glColor3f(0, 0, 1);
    glVertex2f(200, 100);     // top-right

    glColor3f(1, 1, 0);
    glVertex2f(200, -100);    // bottom-right
    glEnd();

    glutSwapBuffers();
}

void idle() {
    glutSetWindow(spinWin);
    glRotatef(0.1, 0, 0, 1);
    glutPostRedisplay();
}

void myInit() {
    glClearColor(1, 1, 1, 1);

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(-250, 250, -250, 250);

    glMatrixMode(GL_MODELVIEW);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_RGB | GLUT_DOUBLE);
    glutInitWindowSize(500, 500);

    glutInitWindowPosition(100, 100);
    glutCreateWindow("Rect 1");
    myInit();
    glutDisplayFunc(display);

    glutInitWindowPosition(600, 100);
    spinWin = glutCreateWindow("Rect 2");
    myInit();
    glutDisplayFunc(display);
    glutIdleFunc(idle);

    glutMainLoop();
    return 0;
}
